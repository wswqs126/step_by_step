# 快排

思路：划分递归，选一个元素，然后划分成两部分，左边的都小于这个元素，右边的都大于这个元素，然后不断递归，最终排好序。

特点：原地排序，不稳定。

需要一个交换函数；需要一个划分求下一个 pivot 位置的过程；需要辅助递归函数.

## 代码实现

```js
var quickSort = arr => {
  // swap fn
  const swap = (array, i, j) => {
    const temp = array[i];
    array[i] = array[j];
    array[j] = temp;
  };

  // partition fn
  const partition = (array, left, right) => {
    // 快排是原地排序，这里的 array 始终都是原始数组，left 和 right 会有变化
    const pivot = array[Math.floor((left + right) / 2)];
    // left 和 right 是数组下标
    let l = left;
    let r = right;

    while (l <= r) {
      while (array[l] < pivot) {
        l++;
      }

      while (array[r] > pivot) {
        r--;
      }

      if (l <= r) {
        swap(array, l, r);
        l++;
        r--;
      }
    }
    
    // 返回 l 是因为，结束后 l 左侧都是小于等于主元，而 l右侧（包括 l）都是大于等于主元，所以这是划分的过程
    // 后面的迭代也是对划分后的数组进行迭代 
    return l;
  };

  // recursive fn
  const quick = (array, left, right) => {
    let index;
    if (array.length > 1) {
      index = partition(array, left, right);

      // 若左边数组不是1个元素，继续递归
      if (left < index - 1) {
        quick(array, left, index - 1);
      }

      if (right > index) {
        quick(array, index, right);
      }
    }
  };

  // 写成了 quick(arr, 0, array.lenth - 1)
  // 1. 变量写错，2，length 写错。。。
  quick(arr, 0, arr.length - 1);
};

var a = [6, 5, 4, 26, 5, 34, 2, 3, 44, 3, 2, 144, 22, 6, 76, 66, 88];
quickSort(a);
console.log(a);
```

# 快排

需要一个交换函数；需要一个划分求下一个pivot位置的过程；需要辅助递归函数.

## 代码实现

```js
var quickSort = arr => {
  // swap fn
  const swap = (array, i, j) => {
    const temp = array[i];
    array[i] = array[j];
    array[j] = temp;
  };

  // partition fn
  const partition = (array, left, right) => {
    const pivot = array[Math.floor((left + right) / 2)];
    let l = left;
    let r = right;

    while (l <= r) {
      while (array[l] < pivot) {
        l++;
      }

      while (array[r] > pivot) {
        r--;
      }

      if (l <= r) {
        swap(array, l, r);
        l++;
        r--;
      }
    }

    return l;
  };

  // recursive fn
  const quick = (array, left, right) => {
    let index;
    if (array.length > 1) {
      index = partition(array, left, right);

      // 若左边数组不是1个元素，继续递归
      if (left < index - 1) {
        quick(array, left, index - 1);
      }

      if (right > index) {
        quick(array, index, right);
      }
    }
  };

  // 写成了 quick(arr, 0, array.lenth - 1)
  // 1. 变量写错，2，length 写错。。。
  quick(arr, 0 , arr.length - 1);
};

var a = [6,5,4,26,5,34,2,3,44,3,2,144,22,6,76,66,88];
quickSort(a);
console.log(a);
```

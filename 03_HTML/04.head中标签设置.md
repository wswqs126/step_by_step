<title>旗下基金</title>
<meta charset="UTF-8">

<!-- 开启 https 默认关闭 a 标签的 dns 预解析，这里手动开启 -->
<meta http-equiv="x-dns-prefetch-control" content="on">
<!-- 指定特定 url 进行 dns 预解析 -->
<link rel="dns-prefetch" href="//static-https.lansheng8.com">

<meta name="viewport" content="width=device-width, initial-scale=1.0, user-scalable=no">

<!--  pc 端：ie8 ～ ie10 有效，使用 edge 引擎渲染，ie11 默认这个选项，chrome=1，不推荐使用了 -->
<meta http-equiv="X-UA-Compatible" content="IE=edge">

<!-- pc 端：国内双核浏览器，优先 webkit 内核渲染 -->
<meta name="renderer" content="webkit">

<!-- 关键词、描述等 -->
<meta name="keywords" content="xx,xx,xx">
<meta name="description" content="xxxxxx.">

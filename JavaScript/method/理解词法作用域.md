涉及词法作用域和动态作用域 http://www.jianshu.com/p/70b38c7ab69c

```js
// code 1
var a = 2;

function foo() {
  console.log(a); // 会输出2还是3？
}

function bar() {
  var a = 3;
  foo();
}
bar();

// code 2
var a = 2;

function bar() {
  var a = 3;
  function foo() {
    console.log(a); // 会输出2还是3？
  }
  foo();
}

bar();
```

code1输出是2，code2输出3 why？。首先，上面的代码调换下顺序：

```js
var a;

function foo() {
  console.log(a); // 会输出2还是3？
}

function bar() {
  var a = 3;
  foo();
}

a = 2;

bar();
```

通过设置 `debugger;` 就可以自动在之后第二行代码开始，开启断点调试。   

```js
function baz() {
  console.log('baz');
  bar();
}

function bar() {
  console.log('bar');
  foo();
}

function foo() {
  // 设置断点
  debugger;
  
  console.log('foo');
}

baz();
```

下面的代码执行有点疑惑：

```js
var a = 0;

function bar() {
  var a = 1;

  console.log('baz', this.a); // 0

  foo(); // 这个为什么也是 0，调用位置不是在 bar？
}

function foo() {
  var a = 2;

  console.log('foo', this.a);
}

bar();
```
